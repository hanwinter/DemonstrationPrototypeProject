## 2）热力映射逻辑（核心）

```markdown
<script setup>
import { computed } from "vue"
import { abnormalities } from "../mock/data"

const props = defineProps({
  gender: String,
  selectedMode: String,
  selectedOrgan: String
})

const severityMap = {
  low: "heat-yellow",
  moderate: "heat-orange",
  high: "heat-red"
}

// organ -> severity
const organMap = computed(() => {
  const map = {}
  abnormalities.forEach(a => {
    map[a.organ] = a.severity
  })
  return map
})

const getClass = (organ) => {
  const sev = organMap.value[organ]
  return sev ? severityMap[sev] : "normal"
}
</script>
```
